function randomName(len) {
    let chars = 'abcdefghijklmnopqrstuvwxyz';
    let out = chars[Math.floor(Math.random() * chars.length)];
    for (let i = 1; i < len; i++) out += (chars + '0123456789_')[Math.floor(Math.random() * 36)];
    return out;
}

function constantFolding(code) {
    let folded = code.replace(/(\d+(?:\.\d+)?)\s*([+\-*/%])\s*(\d+(?:\.\d+)?)/g, (match, a, op, b) => {
        let na = parseFloat(a), nb = parseFloat(b);
        if (isNaN(na) || isNaN(nb)) return match;
        if (op === '+') return (na + nb).toString();
        if (op === '-') return (na - nb).toString();
        if (op === '*') return (na * nb).toString();
        if (op === '/') return nb !== 0 ? (na / nb).toString() : match;
        if (op === '%') return nb !== 0 ? (na % nb).toString() : match;
        return match;
    });
    folded = folded.replace(/if\s+false\s+then[\s\S]*?end/g, '');
    folded = folded.replace(/if\s+true\s+then([\s\S]*?)end/g, '$1');
    return folded;
}

        `do local _x=math.random(1,99) _x=_x%2+_x end`,
        `do local _s='dead' _s=_s:len()>0 and _s or '' end`,
        `do local _t={} _t[1]=1 _t[2]=2 if _t[1]==1 then _t[1]=0 end end`,
        `do local _z=0 while _z<3 do _z=_z+1 end end`,

}

function buildAntiDebugCode(intensity) {
    let methods = [
        `pcall(function() if debug and debug.getinfo then local _={} setmetatable(_,{__index=function() error("") end}) end end)`,
        `pcall(function() if debug and debug.gethook then local h=debug.gethook() if h then error("") end end end)`,
        `pcall(function() local t=tick() for i=1,50000 do end if tick()-t>0.5 then error("") end end)`
    ];
    let count = intensity === 'lite' ? 1 : (intensity === 'medium' ? 2 : 3);
    let code = `-- anti-debug (${intensity})\n`;
    for (let i = 0; i < count; i++) code += methods[i % methods.length] + "\n";
    return code;
}

function encryptString(str) {
    let enc = [], key = [];
    for (let i = 0; i < str.length; i++) {
        let k = Math.floor(Math.random() * 255);
        enc.push((str.charCodeAt(i) + k) % 256);
        key.push(k);
    }
    return { enc, key };
}

    let r2 = Math.floor(Math.random() * 20) + 2;

}

// ======================== COMPILER ========================
class SafeCompiler {
    constructor(src) {
        this.src = src; this.pos = 0; this.tokens = [];
        this.code = []; this.strings = []; this.numbers = [];
        this.vars = {}; this.varc = 0; this.logs = [];
    }
    tokenize() {
        let i = 0, len = this.src.length, toks = [];
        while (i < len) {
            let ch = this.src[i];
            if (ch === ' ' || ch === '\t' || ch === '\r') { i++; continue; }
            if (ch === '\n') { toks.push({ type: 'newline' }); i++; continue; }
            if (ch === '-' && this.src[i+1] === '-') {
                i += 2;
                if (this.src[i] === '[' && this.src[i+1] === '[') {
                    i += 2;
                    while (i < len && !(this.src[i] === ']' && this.src[i+1] === ']')) i++;
                    i += 2;
                } else { while (i < len && this.src[i] !== '\n') i++; }
                continue;
            }
            if (ch === '"' || ch === "'") {
                let quote = ch, str = ''; i++;
                while (i < len && this.src[i] !== quote) {
                    if (this.src[i] === '\\' && i+1 < len) { i++; str += this.src[i]; }
                    else str += this.src[i];
                    i++;
                }
                if (i < len) i++;
                toks.push({ type: 'string', value: str }); continue;
            }
            if (/[0-9]/.test(ch)) {
                let num = '';
                while (i < len && /[0-9.]/.test(this.src[i])) num += this.src[i++];
                toks.push({ type: 'number', value: parseFloat(num) }); continue;
            }
            if (/[a-zA-Z_]/.test(ch)) {
                let name = '';
                while (i < len && /[a-zA-Z0-9_]/.test(this.src[i])) name += this.src[i++];
                let kw = new Set(['local','if','then','else','elseif','for','do','end','print','while','repeat','until','return','nil','true','false','and','or','not']);
                toks.push({ type: kw.has(name) ? 'keyword' : 'identifier', value: name }); continue;
            }
            let two = ch + (this.src[i+1] || '');
            if (['..','==','~=','<=','>='].includes(two)) { toks.push({ type: 'operator', value: two }); i += 2; continue; }
            if ('+-*/%<>=.,;:(){}[]#^'.includes(ch)) { toks.push({ type: 'symbol', value: ch }); i++; continue; }
            i++;
        }
        this.tokens = toks;
        this.tokens.push({ type: 'eof' });
    }
    peek() { return this.tokens[this.pos] || { type: 'eof' }; }
    next() { return this.tokens[this.pos++] || { type: 'eof' }; }
    emit(op, arg) { this.code.push(op); if (arg !== undefined) this.code.push(arg); }
    getVar(name) { if (this.vars[name] === undefined) this.vars[name] = this.varc++; return this.vars[name]; }
    addNum(n) { let idx = this.numbers.indexOf(n); if (idx === -1) { idx = this.numbers.length; this.numbers.push(n); } return idx; }
    addStr(s) { let idx = this.strings.indexOf(s); if (idx === -1) { idx = this.strings.length; this.strings.push(s); } return idx; }
    parseExpr() { this.parseOr(); }
    parseOr() { this.parseAnd(); while (this.peek().type === 'keyword' && this.peek().value === 'or') { this.next(); this.parseAnd(); } }
    parseAnd() { this.parseCmp(); while (this.peek().type === 'keyword' && this.peek().value === 'and') { this.next(); this.parseCmp(); } }
    parseCmp() {
        this.parseConcat();
        let ops = { '==': OP.EQ, '~=': OP.NE, '<': OP.LT, '>': OP.GT, '<=': OP.LE, '>=': OP.GE };
        while (this.peek().type === 'operator' && ops[this.peek().value]) {
            let op = ops[this.next().value]; this.parseConcat(); this.emit(op);
        }
    }
    parseConcat() {
        this.parseAddSub();
        while (this.peek().type === 'operator' && this.peek().value === '..') {
            this.next(); this.parseAddSub(); this.emit(OP.CONCAT);
        }
    }
    parseAddSub() {
        this.parseMulDiv();
        while (this.peek().type === 'symbol' && (this.peek().value === '+' || this.peek().value === '-')) {
            let op = this.next().value === '+' ? OP.ADD : OP.SUB;
            this.parseMulDiv(); this.emit(op);
        }
    }
    parseMulDiv() {
        this.parseUnary();
        while (this.peek().type === 'symbol' && '*/%'.includes(this.peek().value)) {
            let op = { '*': OP.MUL, '/': OP.DIV, '%': OP.MOD }[this.next().value];
            this.parseUnary(); this.emit(op);
        }
    }
    parseUnary() {
        if (this.peek().type === 'symbol' && this.peek().value === '-') {
            this.next(); this.parsePrimary();
            this.emit(OP.PUSH_NUM, this.addNum(0)); this.emit(OP.SUB);
        } else if (this.peek().type === 'keyword' && this.peek().value === 'not') {
            this.next(); this.parsePrimary(); this.emit(OP.NOT);
        } else this.parsePrimary();
    }
    parsePrimary() {
        let t = this.peek();
        if (t.type === 'number') { this.emit(OP.PUSH_NUM, this.addNum(t.value)); this.next(); }
        else if (t.type === 'string') { this.emit(OP.PUSH_STR, this.addStr(t.value)); this.next(); }
        else if (t.type === 'identifier') {
            let name = this.next().value;
            if (this.peek().type === 'symbol' && this.peek().value === '(') {
                this.next();
                let args = 0;
                if (this.peek().value !== ')') {
                    this.parseExpr(); args++;
                    while (this.peek().value === ',') { this.next(); this.parseExpr(); args++; }
                }
                if (this.peek().value === ')') this.next();
                this.emit(OP.CALL, this.addStr(name));
                this.emit(args);
            } else { this.emit(OP.LOAD, this.getVar(name)); }
        } else if (t.type === 'symbol' && t.value === '(') {
            this.next(); this.parseExpr();
            if (this.peek().value === ')') this.next();
        } else if (t.type === 'keyword' && (t.value === 'nil' || t.value === 'false')) {
            this.emit(OP.PUSH_NUM, this.addNum(0)); this.next();
        } else if (t.type === 'keyword' && t.value === 'true') {
            this.emit(OP.PUSH_NUM, this.addNum(1)); this.next();
        } else this.next();
    }
    expect(type, val) {
        let t = this.peek();
        if (t.type !== type || (val !== undefined && t.value !== val)) throw new Error(`Expected ${val||type}, got ${t.value||t.type}`);
        return this.next();
    }
    parseBlock(stop) {
        while (this.peek().type !== 'eof') {
            if (stop.includes(this.peek().value)) break;
            this.parseStatement();
            while (this.peek().type === 'newline') this.next();
        }
    }
    parseStatement() {
        let t = this.peek();
        if (t.type === 'newline') { this.next(); return; }
        if (t.type === 'keyword') {
            switch(t.value) {
                case 'local': {
                    this.next();
                    let v = this.expect('identifier').value;
                    let idx = this.getVar(v);
                    if (this.peek().value === '=') { this.next(); this.parseExpr(); this.emit(OP.STORE, idx); }
                    else { this.emit(OP.PUSH_NUM, this.addNum(0)); this.emit(OP.STORE, idx); }
                    break;
                }
                case 'print': {
                    this.next(); this.expect('symbol', '(');
                    if (this.peek().value !== ')') {
                        this.parseExpr();
                        while (this.peek().value === ',') { this.next(); this.parseExpr(); }
                    }
                    if (this.peek().value === ')') this.next();
                    this.emit(OP.PRINT);
                    break;
                }
                case 'if': this.parseIf(); break;
                case 'for': this.parseFor(); break;
                case 'while': this.parseWhile(); break;
                case 'repeat': this.parseRepeat(); break;
                case 'return': {
                    this.next();
                    if (this.peek().value !== 'end' && this.peek().type !== 'newline' && this.peek().type !== 'eof') {
                        this.parseExpr(); this.emit(OP.RET);
                    } else {
                        this.emit(OP.PUSH_NUM, this.addNum(0)); this.emit(OP.RET);
                    }
                    break;
                }
                default: while (this.peek().type !== 'eof' && this.peek().type !== 'newline') this.next();
            }
        } else if (t.type === 'identifier') {
            let name = this.next().value;
            if (this.peek().value === '=') { this.next(); this.parseExpr(); this.emit(OP.STORE, this.getVar(name)); }
            else { while (this.peek().type !== 'eof' && this.peek().type !== 'newline') this.next(); }
        } else this.next();
    }
    parseIf() {
        this.next(); this.parseExpr();
        this.emit(OP.JZ, 0); let jzPos = this.code.length - 1;
        this.expect('keyword', 'then');
        this.parseBlock(['else','elseif','end']);
        if (this.peek().value === 'else') {
            this.next();
            this.emit(OP.JMP, 0); let jmpPos = this.code.length - 1;
            this.code[jzPos] = this.code.length;
            this.parseBlock(['end']);
            if (this.peek().value === 'end') this.next();
            this.code[jmpPos] = this.code.length;
        } else {
            if (this.peek().value === 'end') this.next();
            this.code[jzPos] = this.code.length;
        }
    }
    parseFor() {
        this.next();
        let varN = this.expect('identifier').value;
        let idx = this.getVar(varN);
        this.expect('symbol', '=');
        this.parseExpr(); this.emit(OP.STORE, idx);
        this.expect('symbol', ',');
        this.parseExpr();
        let endIdx = this.getVar('_end_' + varN); this.emit(OP.STORE, endIdx);
        let stepIdx = this.getVar('_step_' + varN);
        if (this.peek().value === ',') { this.next(); this.parseExpr(); this.emit(OP.STORE, stepIdx); }
        else { this.emit(OP.PUSH_NUM, this.addNum(1)); this.emit(OP.STORE, stepIdx); }
        this.expect('keyword', 'do');
        let loopStart = this.code.length;
        this.emit(OP.LOAD, idx); this.emit(OP.LOAD, endIdx); this.emit(OP.LE);
        this.emit(OP.JZ, 0); let jz = this.code.length - 1;
        this.parseBlock(['end']);
        if (this.peek().value === 'end') this.next();
        this.emit(OP.LOAD, idx); this.emit(OP.LOAD, stepIdx); this.emit(OP.ADD); this.emit(OP.STORE, idx);
        this.emit(OP.JMP, loopStart);
        this.code[jz] = this.code.length;
    }
    parseWhile() {
        this.next();
        let loopStart = this.code.length;
        this.parseExpr();
        this.emit(OP.JZ, 0); let jz = this.code.length - 1;
        this.expect('keyword', 'do');
        this.parseBlock(['end']);
        if (this.peek().value === 'end') this.next();
        this.emit(OP.JMP, loopStart);
        this.code[jz] = this.code.length;
    }
    parseRepeat() {
        this.next();
        let start = this.code.length;
        this.parseBlock(['until']);
        this.expect('keyword', 'until');
        this.parseExpr();
        this.emit(OP.JZ, start);
    }
    compile() {
        try {
            this.tokenize();
            while (this.peek().type !== 'eof') {
                this.parseStatement();
                while (this.peek().type === 'newline') this.next();
            }
            this.emit(OP.HALT);
        } catch(e) {
            this.logs.push("parse warning: " + e.message);
            this.emit(OP.HALT);
        }
        return { code: this.code, strings: this.strings, numbers: this.numbers, varCount: this.varc, logs: this.logs };
    }
}

// ======================== POLYMORPHIC DISPATCHER ========================
function buildPolymorphicDispatcher(opCount, seed) {
    let mapping = new Array(opCount);
    let used = new Set();
    for (let op = 0; op < opCount; op++) {
        let mapped, tries = 0;
        do {
            mapped = ((op * seed + 31 + tries * 7) % 200 + 10) & 0xFF;
            tries++;
        } while (used.has(mapped) && tries < 1000);
        used.add(mapped);
        mapping[op] = mapped;
    }
    let inverse = new Array(256).fill(-1);
    for (let op = 0; op < opCount; op++) {
        if (mapping[op] !== undefined) inverse[mapping[op]] = op;
    }
    return { mapping, inverse };
}

function encodeInstruction(encodedOp, arg) {
    // hi = encoded opcode, lo = arg (keduanya <= 32-bit aman di Lua)
    let hi = encodedOp & 0xFFFF;
    let lo = arg & 0xFFFFFFFF;
    return { hi, lo };
}

function buildProfessionalVM(compiled, useXOR, useAD, adIntensity, useCFF, useDeadCode, useStringPoly, useConstEnc, useCFI, useOpaque, useMultiStack, usePolyDispatch, useFastVM) {
    let { code, strings, numbers } = compiled;
    if (code.length === 0) return "-- empty script";

    let polySeed = usePolyDispatch ? Math.floor(Math.random() * 10000) + 1000 : 0;
    let polyConfig = usePolyDispatch ? buildPolymorphicDispatcher(50, polySeed) : null;

    // Build final bytecode sebagai pasangan hi, lo (BUG FIX #4)
    let hiArr = [], loArr = [];
    let i = 0;
    while (i < code.length) {
        let op = code[i];
        let arg = 0;
        if (HAS_ARG.has(op) && i + 1 < code.length) {
            i++;
            arg = code[i];
        }
        // BUG FIX #2: kalau polyDispatch off, encodedOp = op langsung

}

function buildHybrid(script, useXOR, useAD, adIntensity, useCFF, useDeadCode, useOpaque, useConstEnc) {
    let lua = "";
    if (useAD) lua += buildAntiDebugCode(adIntensity) + "\n";
    if (useDeadCode) lua += generateDeadBlock() + generateDeadBlock();
    if (useOpaque) { lua += injectOpaqueBlock(); lua += injectOpaqueBlock(); }
    let enc = Array.from(new TextEncoder().encode(script));
    let finalEnc = enc;
    lua += luaXorManual() + "\n";
    if (useXOR) {
        let key = Array.from({length: 16}, () => Math.floor(Math.random() * 255));
        finalEnc = enc.map((b, i) => b ^ key[i % key.length]);
        lua += `local key={${key.join(',')}}\n`;
        lua += `local enc={${finalEnc.join(',')}}\n`;
        lua += `for i=1,#enc do enc[i]=__xor(enc[i],key[(i-1)%#key+1]) end\n`;
    } else {
        lua += `local enc={${finalEnc.join(',')}}\n`;
    }
    lua += `local strTbl={} for i=1,#enc do strTbl[i]=string.char(enc[i]) end local s=table.concat(strTbl) local fn,err=loadstring(s) if fn then pcall(fn) else warn(err) end`;
    if (useCFF) {
        let steps = 3, stepFn = randomName(6), stVar = randomName(4);
        lua = `local function ${stepFn}(s) ` + Array(steps).fill().map((_,i) => `if s==${i} then return ${i+1} end`).join(' ') + ` if s==${steps} then (function() ${lua} end)() return ${steps+1} end return -1 end local ${stVar}=0 while ${stVar}>=0 do ${stVar}=${stepFn}(${stVar}) end`;
    } else {
        lua = `(function() ${lua} end)()`;
    }
    return lua;
}

function parseMacroBlocks(src) {
    let lines = src.split('\n');
    let blocks = [];
    let currentBlock = { type: 'default', content: [] };
    let macroPatterns = [
        { pattern: /^--!nurhub\s+VM/i, type: 'vm' },
        { pattern: /^--!nurhub\s+ENC/i, type: 'enc' },
        { pattern: /^--!nurhub\s+OMIT/i, type: 'omit' }
    ];
    for (let line of lines) {
        let matched = false;
        for (let mp of macroPatterns) {
            if (mp.pattern.test(line)) {
                if (currentBlock.content.length > 0) blocks.push(currentBlock);
                currentBlock = { type: mp.type, content: [] };
                matched = true; break;
            }
        }
        if (!matched) currentBlock.content.push(line);
    }
    if (currentBlock.content.length > 0) blocks.push(currentBlock);
    return blocks;
}
